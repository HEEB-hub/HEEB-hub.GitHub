133 
134 bool automaticLowDimension::update() //automatic method need update
135 
136 {
137  if (ld_Error_<ld_LimitError_)
138  {
139  Info<<"Finish step "<<Step<<nl;
140  Info<<"Relative error is now less than the creterion! : "
141  <<ld_LimitError_.value()*100<<"%"<<endl
142  <<"Iteration finished!"<<endl<<endl<<endl<<endl<<endl;
143  return false;
144  }
145 
146  else if (Step>=ld_maxStep_.value())
147  {
148  Info<<"Finish step "<<Step<<nl;
149  Info<<"Reach the max step limit! "
150  <<"Iteration finished!"<<endl<<endl<<endl<<endl<<endl;
151  return false;
152  }
153 
154  else
155  {
156  int max=0;
157  for (int i=1;i<3;i++)
158  if (fieldVector.value()[max]<fieldVector.value()[i])
159  max=i;
160  const vector normalsize = fieldVector.value()/fieldVector.value()[max];
161  const vector oldsize=ld_Dimension_.value();
162  do
163  {
164  Info<<"Finish step "<<Step<<", getting a new lowDimensionSet... "<<nl
165  <<"Checking if it is available..."<<endl<<nl;
166  vector temp(normalsize*Step*ld_updateRatio_.value());
167  ld_Dimension_.value() = vector(
168  int (0.5+temp.x()+ld_Initial_set_.value().x()),
169  int (0.5+temp.y()+ld_Initial_set_.value().y()),
170  int (0.5+temp.z()+ld_Initial_set_.value().z())
171  );
172  Step++;
173  if (Step>ld_maxStep_.value())
174  { Info<<"No new lowDimensionSet available"<<nl
175  <<"Reach the max step limit! "
176  <<"Iteration finished!"<<endl<<endl<<endl<<endl<<endl;
177  return false;
178  }
179  }
180  while (ld_Dimension_.value()[0]==oldsize[0] &&
181  ld_Dimension_.value()[1]==oldsize[1] &&
182  ld_Dimension_.value()[2]==oldsize[2] );
183  Valid_step++;
184  Info<<"The lowDimensionSet updated successfully!"<<endl
185  <<"Going to start next iteration!"<<endl<<endl<<endl<<endl<<endl;
186  fieldNums();
187  setPoints();
188  return true;
189  }
190 }
