133 void lowDimensionMethod::fieldRange()
134 {
135  const pointField & pointlist = mesh_.points();
136 
137  scalar xMax(max(pointlist & vector(1,0,0)));
138  scalar yMax(max(pointlist & vector(0,1,0)));
139  scalar zMax(max(pointlist & vector(0,0,1)));
140 
141  dimensionedVector maxvector("maxvector",mesh_.C().dimensions(),vector(xMax,yMax,zMax));
142 
143  scalar xMin(min(pointlist & vector(1,0,0)));
144  scalar yMin(min(pointlist & vector(0,1,0)));
145  scalar zMin(min(pointlist & vector(0,0,1)));
146 
147  dimensionedVector minvector("minvector",mesh_.C().dimensions(),vector(xMin,yMin,zMin));
148  dimensionedVector fieldvector("fieldvector",mesh_.C().dimensions(),maxvector.value()-minvector.value());
149 
150  minVector = minvector;
151  fieldVector = fieldvector;
152 }

158 void lowDimensionMethod::setPoints()
159 {
160  const vector subpoint_num(
161  ld_Dimension_.value().x()+1,
162  ld_Dimension_.value().y()+1,
163  ld_Dimension_.value().z()+1);
164 
165  const vector subzone_len(
166  fieldVector.value().x()/ld_Dimension_.value().x(),
167  fieldVector.value().y()/ld_Dimension_.value().y(),
168  fieldVector.value().z()/ld_Dimension_.value().z());
169 
170  for(int i=0;i<3;i++)
171  {
172  ld_cutpoint_[i].resize(subpoint_num[i]);
173  for(unsigned int j=0;j<ld_cutpoint_[i].size();j++)
174  {
175  ld_cutpoint_[i][j]=j*subzone_len[i];
176  }
177  }
178 }

184  //- low-dimension process of scalarfield
185 void lowDimensionMethod::ldProcess()
186  {
187  Info<<"Starting the Low-Dimension Dividing ...please wait"<<endl<<endl;
188 
189  dimensionedScalar* totalv =new dimensionedScalar[ld_fieldNums_];
190  dimensionedScalar* totalvc=new dimensionedScalar[ld_fieldNums_];
191 
192  forAll (C_,celli)
193  {
194  const vector abs_coord=mesh_.C()[celli]-minVector.value();
195  vector temp_pos(0,0,0);
196  for(int i=0;i<3;i++)
197  {
198  for(unsigned int j=0;j<ld_Dimension_.value()[i];j++)
199  {
200  if(abs_coord[i] > ld_cutpoint_[i][j] &&
201  abs_coord[i] <= ld_cutpoint_[i][j+1] )
202  {
203  temp_pos[i] = j;
204  break;
205  }
206  }
207  }
208  label ZoneLabel=temp_pos[0]+temp_pos[1]*ld_Dimension_.value().x()+temp_pos[2]*ld_Dimension_.value().x()*ld_Dimension_.value().y();
209  totalv[ZoneLabel] += mesh_.V()[celli];
210  totalvc[ZoneLabel] += mesh_.V()[celli]*C_[celli];
211  }
212 
213  for (int i=0;i<ld_fieldNums_;i++)
214  {
215  ld_Scalar_[i]=totalvc[i]/totalv[i];
216  }
217 
218  Info<<"Dividing into "<<ld_fieldNums_<<" zones successfully, with "
219  <<ld_Dimension_.value()<<" pieces in each (X,Y,Z) coordinate!"<<endl<<endl;
220 
221  delete [] totalv;
222  delete [] totalvc;
223  }

226  //- low-dimension error calculation
227 void lowDimensionMethod::ldError()
228  {
229 
230  Info<<"Calculating the Low-Dimension Error ...please wait"<<endl<<endl;
231  dimensionedScalar totalError2(0);
232  dimensionedScalar totalZone2(0);
233 
234  dimensionedScalar* totalv = new dimensionedScalar[ld_fieldNums_];
235  dimensionedScalar* subError2 = new dimensionedScalar[ld_fieldNums_];
236  dimensionedScalar* subZone2 = new dimensionedScalar[ld_fieldNums_];
237 
238  dimensionedScalar errorVar(0);
239  dimensionedScalar totalV(0);
240 
241  forAll (C_,celli)
242  {
243  const vector abs_coord=mesh_.C()[celli]-minVector.value();
244  vector temp_pos(0,0,0);
245  for(int i=0;i<3;i++)
246  {
247  for(unsigned int j=0;j<(ld_cutpoint_[i].size()-1);j++)
248  {
249  if(abs_coord[i] > ld_cutpoint_[i][j] &&
250  abs_coord[i] <= ld_cutpoint_[i][j+1] )
251  {
252  temp_pos[i] = j;
253  break;
254  }
255  }
256  }
257  label ZoneLabel=temp_pos[0]+temp_pos[1]*ld_Dimension_.value().x()+temp_pos[2]*ld_Dimension_.value().x()*ld_Dimension_.value().y();
258 
259  totalError2 += mesh_.V()[celli]*pow(ld_Scalar_[ZoneLabel]-C_[celli],2);
260  totalZone2 += mesh_.V()[celli]*pow(C_[celli],2);
261 
262  totalv[ZoneLabel] += mesh_.V()[celli];
263  subError2[ZoneLabel] += mesh_.V()[celli]*pow(ld_Scalar_[ZoneLabel]-C_[celli],2);
264  subZone2[ZoneLabel] += mesh_.V()[celli]*pow(C_[celli],2);
265 
266  // totalError2 += pow(ld_Scalar_[ZoneLabel]-C_[celli],2);
267  // totalZone2 += pow(C_[celli],2);
268  }
269 
270  ld_Error_=sqrt(totalError2/totalZone2);
271 
272  for (int i=0;i<ld_fieldNums_;i++)
273  {
274  ld_SubError_[i]=sqrt(subError2[i]/subZone2[i]);
275  errorVar += totalv[i]*pow(ld_SubError_[i]-ld_Error_,2);
276  totalV += totalv[i];
277  }
278 
279  errorVar=sqrt(errorVar/totalV);
280 
281  Info<<"The Low-Dimension Discretization Error of "<<ld_Dimension_.value()<<" set equals "<<ld_Error_.value()*100<<"%"<<endl<<endl;
282  Info<<"The Low-Dimension Discretization Error variance of "<<ld_Dimension_.value()<<" set equals "<<errorVar.value()*100<<"%"<<endl<<endl;
283 
284  delete [] totalv;
285  delete [] subError2;
286  delete [] subZone2;
287  }
