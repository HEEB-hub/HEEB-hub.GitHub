100 void nonUniformMethod::genMeshSizeL()
101 {
102  const faceList & ff = mesh_.faces();//mesh faces presented as 4 points;
103  const pointField & pp = mesh_.points(); //mesh points' specific (x,y,z);
104  std::vector < std::vector<double> > mlen(3);
105  forAll (mesh_.C(), celli)
106  {
107  const cell & cc = mesh_.cells()[celli];//single mesh presented as 6 faces
108  labelList pLabels(cc.labels(ff));//single mesh presented as points' label, important
109  pointField pLocal(pLabels.size(), vector::zero);
110 
111  forAll (pLabels, pointi) //find the (x,y,z) of every point of mesh
112  {
113  pLocal[pointi] = pp[pLabels[pointi]];
114  }
115  scalar xDim=Foam::max(pLocal & vector(1,0,0)) - Foam::min(pLocal & vector(1,0,0));
116  scalar yDim=Foam::max(pLocal & vector(0,1,0)) - Foam::min(pLocal & vector(0,1,0));
117  scalar zDim=Foam::max(pLocal & vector(0,0,1)) - Foam::min(pLocal & vector(0,0,1));
118  vector Dim(xDim,yDim,zDim);
119  for(int i=0;i<3;i++)
120  mlen[i].push_back(Dim[i]);
121  }
122  vector nDimMinMeshSize_(vector::zero);
123  for(int i=0;i<3;i++)
124  {
125  auto maxPosition = max_element(mlen[i].begin(), mlen[i].end());
126  auto minPosition = min_element(mlen[i].begin(), mlen[i].end());
127  nDimMaxMeshSize_[i] = *maxPosition/fieldVector.value()[i];
128  nDimMinMeshSize_[i] = *minPosition/fieldVector.value()[i];
129  Info<<"nonDimension Max Mesh Size (percent of domain length) in "
130  << char('x'+i)<<" direction: "<< nDimMaxMeshSize_[i] << nl;
131  Info<<"nonDimension Min Mesh Size (percent of domain length) in "
132  << char('x'+i)<<" direction: "<< nDimMinMeshSize_[i] << nl <<nl;
133  }
134  Info<<"nonDimension Mesh Size above for setting reference!\n"
135  <<"Press 'Enter' to continue!"<<nl;
136  cin.get();
137 }

140 void nonUniformMethod::genSubLayerC()
141 {
142  for(int i=0;i<3;i++)
143  {
144  double nDimLay_Thick_= nDimMaxMeshSize_[i] * nDimSubLayerThick_[i];
145  sublayer_Num_[i] = int (pow(nDimLay_Thick_ , -1));
146  double minLayNum_temp = nDimMinZoneThick_[i] / (1/double(sublayer_Num_[i]));
147  least_LayerNum_[i] = int (minLayNum_temp);
148  if (minLayNum_temp - least_LayerNum_[i] > 0 || least_LayerNum_[i]==0)
149  least_LayerNum_[i]++;
150  sublayer_Thick[i] = fieldVector.value()[i] / sublayer_Num_[i];
151  dimensionedScalar* laytotalv = new dimensionedScalar[sublayer_Num_[i]];
152  dimensionedScalar* laytotalvc = new dimensionedScalar[sublayer_Num_[i]];
153  forAll (C_,celli)
154  {
155  const double abs_pos = mesh_.C()[celli][i]-minVector.value()[i];
156  int temp_pos = int (abs_pos / sublayer_Thick[i]) ;
157  laytotalv [temp_pos] += mesh_.V()[celli];
158  laytotalvc [temp_pos] += mesh_.V()[celli] * C_[celli];
159  }
160  for (int j=0;j<sublayer_Num_[i];j++)
161  {
162  if (!i) total_volume_ += laytotalv[j].value();
163  if (laytotalv[j].value() != 0 )
164  {
165  sublayer_V_[i].push_back ( laytotalv[j].value() );
166  sublayer_Scalar_[i].push_back ( (laytotalvc[j] / laytotalv[j]).value() );
167  }
168  else
169  {
170  Info<<"Too small 'subLayerThick' set for present mesh!"<<nl
171  <<"Please check and enlarge!"<<nl;
172  std::abort();
173  }
174  }
175  Info<<"Dividing into "<<sublayer_Num_[i]<<" sub-layers in "
176  << char('x'+i) <<" direction successfully! "<<nl
177  <<"The minimun sublayer number is "<< least_LayerNum_[i] <<nl<<nl;
178  delete [] laytotalv;
179  delete [] laytotalvc;
180  }
181  checkSet();
182 }

210 inline void nonUniformMethod::accura_isbetter( std::vector<int> combine_set ,const int i)
211 {
212  int k = least_LayerNum_[i];
213  const int m = ld_Dimension_.value()[i]-1;
214  scalar temp_error=0;
215  scalar temp_v[m+1]{};
216  scalar temp_c[m+1]{};
217  scalar temp_e[m+1]{};
218  combine_set.push_back(sublayer_Num_[i]);
219  for (int kk=1; kk<m+1; kk++)
220  combine_set[kk] += kk*(k-1);
221  for (int lay=0; lay < sublayer_Num_[i]; lay++)
222  {
223  for (int p=0; p < m+1; p++)
224  {
225  if (lay >= combine_set[p] && lay < combine_set[p+1])
226  {
227  temp_v[p] += sublayer_V_[i][lay];
228  temp_c[p] += sublayer_V_[i][lay]*sublayer_Scalar_[i][lay];
229  break;
230  }
231  }
232  }
233  for (int p=0; p < m+1; p++)
234  temp_c[p] = temp_c[p] / temp_v[p];
235  for (int lay=0; lay < sublayer_Num_[i]; lay++)
236  {
237  for (int p=0; p < m+1; p++)
238  {
239  if (lay >= combine_set[p] && lay < combine_set[p+1])
240  {
241  temp_e[p] += sublayer_V_[i][lay]*pow(sublayer_Scalar_[i][lay]-temp_c[p],2);
242  break;
243  }
244  }
245  }
246  for (int p=0; p < m+1; p++)
247  {
248  temp_e[p] = Foam::sqrt(temp_e[p]/temp_v[p]);
249  temp_error += temp_e[p]*temp_v[p]/total_volume_;
250  }
251  if (best_error[i]==0) best_error[i] = temp_error;
252  if (temp_error < best_error[i])
253  {
254  best_error[i]=temp_error;
255  for(unsigned int j=0;j<ld_cutpoint_[i].size();j++)
256  ld_cutpoint_[i][j] = combine_set[j]*sublayer_Thick[i];
257  }
258 }

261 bool nonUniformMethod::accuracyPoint() //realization of the 'precise sub-layer recombination' algorithm to find the accuracy cut point
262 {
263 checkSet();
264 best_error=vector::zero;
265  if( priority_ == "accuracy" )
266  {
267  for(int i=0;i<3;i++)
268  {
269  const int k = least_LayerNum_[i];
270  const int m = ld_Dimension_.value()[i]-1;
271  const int n = sublayer_Num_[i]-ld_Dimension_.value()[i]*(k-1)-1;
272  std::vector<int> temp_comb(m+1);
273  temp_comb[0]=0;
274  int level(2);
275  const int maxlevel(m);
276  const int volumn(n-m);
277  if (m==0) continue;
278  if (m==1)
279  {
280  for(int j=1; j < n+1; j++)
281  {
282  temp_comb[1]=j;
283  accura_isbetter(temp_comb,i);
284  }
285  }
286  else
287  {
288  temp_comb[1]=1;
289  while(1)
290  {
291  temp_comb[level] = temp_comb[level-1] + 1;
292  if(level==maxlevel)
293  {
294  accura_isbetter(temp_comb,i);
295  for(int j=temp_comb[level]+1; j<n+1; j++)
296  {
297  temp_comb[level] = j;
298  accura_isbetter(temp_comb,i);
299  }
300  level--;
301  while(temp_comb[level]==level+volumn && level>=1)
302  level--;
303  if(level<1) break;
304  temp_comb[level]++;
305  }
306  level++;
307  }
308  }
309  for(unsigned int j=0;j<ld_cutpoint_[i].size();j++)
310  {
311  Info <<"Optimal cutpoint in "<< char('x'+i) << " direction: "
312  << ld_cutpoint_[i][j] << nl;
313  }
314  Info <<nl<<nl;
315  }
316  return true;
317  }
318  else
319  return false;
320 }

323 int nonUniformMethod::fast_spaceFind(std::vector<int> input_cut, const int i)
324 {
325  int m = input_cut.size()-1;
326  int space_find=0;
327  scalar temp_v[m]{};
328  scalar temp_c[m]{};
329  scalar temp_e[m]{};
330  for (int lay=0; lay < sublayer_Num_[i]; lay++)
331  {
332  for (int p=0; p < m; p++)
333  {
334  if (lay >= input_cut[p] && lay < input_cut[p+1])
335  {
336  temp_v[p] += sublayer_V_[i][lay];
337  temp_c[p] += sublayer_V_[i][lay]*sublayer_Scalar_[i][lay];
338  //Hx=();
339  break;
340  }
341  }
342  }
343  for (int p=0; p < m; p++)
344  temp_c[p] = temp_c[p] / temp_v[p];
345  for (int lay=0; lay < sublayer_Num_[i]; lay++)
346  {
347  for (int p=0; p < m; p++)
348  {
349  if (lay >= input_cut[p] && lay < input_cut[p+1])
350  {
351  temp_e[p] += sublayer_V_[i][lay]*pow(sublayer_Scalar_[i][lay]-temp_c[p],2);
352  break;
353  }
354  }
355  }
356  for (int p=0; p < m; p++)
357  {
358  temp_e[p] = Foam::sqrt(temp_e[p]/temp_v[p]);
359  if (p>0)
360  if (temp_e[p]>temp_e[space_find] &&
361  input_cut[p+1]-input_cut[p]>=2*least_LayerNum_[i])
362  space_find=p;
363  }
364  return space_find;
365 }

368 int nonUniformMethod::fast_pointFind(int start, int end, const int i)
369 {
370  int point_find=start+least_LayerNum_[i];
371  double min_error=0;
372  for (int j=start+least_LayerNum_[i];j<=end-least_LayerNum_[i];j++)
373  {
374  scalar temp_error=0;
375  scalar temp_v[2]{};
376  scalar temp_c[2]{};
377  scalar temp_e[2]{};
378  for (int lay=start; lay < end; lay++)
379  {
380  if (lay<j)
381  {
382  temp_v[0] += sublayer_V_[i][lay];
383  temp_c[0] += sublayer_V_[i][lay]*sublayer_Scalar_[i][lay];
384  }
385  else
386  {
387  temp_v[1] += sublayer_V_[i][lay];
388  temp_c[1] += sublayer_V_[i][lay]*sublayer_Scalar_[i][lay];
389  }
390  }
391 
392  for (int p=0; p < 2; p++)
393  temp_c[p] = temp_c[p] / temp_v[p];
394 
395  for (int lay=start; lay < end; lay++)
396  {
397  if (lay<j)
398  temp_e[0] += sublayer_V_[i][lay]*pow(sublayer_Scalar_[i][lay]-temp_c[0],2);
399  else
400  temp_e[1] += sublayer_V_[i][lay]*pow(sublayer_Scalar_[i][lay]-temp_c[1],2);
401  }
402 
403  for (int p=0; p < 2; p++)
404  {
405  temp_e[p] = Foam::sqrt(temp_e[p]/temp_v[p]);
406  temp_error += temp_e[p]*temp_v[p]/(temp_v[0]+temp_v[1]);
407  }
408  if (min_error==0)
409  min_error = temp_error;
410  if (temp_error < min_error)
411  {
412  min_error = temp_error;
413  point_find=j;
414  }
415  }
416  return point_find;
417 }

420 bool nonUniformMethod::fastPoint() //a dividing method based on the information entropy evaluation.
421 {
422 checkSet();
423  if( priority_ == "fast" )
424  {
425  for (int i=0;i<3;i++)
426  {
427  std::vector<int> temp_point{0,sublayer_Num_[i]};
428  while (temp_point.size()<ld_Dimension_.value()[i]+1)
429  {
430  int cutSpace=fast_spaceFind(temp_point, i);
431  int cutPoint=fast_pointFind(temp_point[cutSpace], temp_point[cutSpace+1], i);
432  temp_point.push_back(cutPoint);
433  sort(temp_point.begin(), temp_point.end());
434  }
435  for(unsigned int j=0;j<ld_cutpoint_[i].size();j++)
436  {
437  Info << "temp_point[" << j <<"]: "<< temp_point[j]<<nl;
438  ld_cutpoint_[i][j] = temp_point[j]*sublayer_Thick[i];
439  }
440  }
441  return true;
442  }
443  else
444  return false;
445 }
